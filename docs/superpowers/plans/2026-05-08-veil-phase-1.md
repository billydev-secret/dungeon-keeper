> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the Veil module's data layer and image crop pipeline — DB schema (migration 009), models, repo functions, NudeNet/face detection, crop rendering, and a stub cog registration. No Discord slash commands yet; Phase 2 adds those.

**Architecture:** Flat service files (`services/veil_*.py`), one migration SQL file (`migrations/009_veil.sql`), and a stub cog (`cogs/veil_cog.py`). The pipeline processes NSFW images through NudeNet ONNX detection → mediapipe face exclusion → PIL crop in memory; original images are never persisted beyond a temp file used by NudeNet. All heavy imports (nudenet, mediapipe, PIL) are lazy — imported inside function bodies only — so `import services.veil_pipeline` is always safe and never triggers model loads.

**Tech Stack:** sqlite3 sync (existing project pattern), nudenet 3.4.x (ONNX CPU), mediapipe 0.10.14 (pinned — breaking changes between minors), Pillow 10+, numpy 1.26+; pytest + pytest-asyncio for tests.

## Codebase Conventions to Know

- **DB**: sync `sqlite3` via `open_db(db_path)` context manager from `db_utils.py`. WAL mode. `row_factory=sqlite3.Row`. No aiosqlite in production code.
- **Migrations**: numbered `.sql` files in `migrations/`, applied by `apply_migrations_sync(db_path)`. Use `CREATE TABLE IF NOT EXISTS` and `CREATE INDEX IF NOT EXISTS`. Next number: `009`.
- **Config storage**: existing `config` table `(guild_id, key, value)`. Helpers: `get_config_value(conn, key, default, guild_id)` and `set_config_value(conn, key, value, guild_id)` from `db_utils.py`. Key convention: `veil_role_id`, `veil_channel_id`, etc. (underscore-prefixed, like `birthday_channel_id`).
- **Cog pattern**: `class XCog(commands.Cog)` with `def __init__(self, bot, ctx)` + `async def setup(bot)`.
- **Tests**: no `__init__.py` in test subdirs. `tests/unit/`, `tests/components/`, `tests/cogs/`. `sync_db_path` fixture does not exist yet — added in Task 4.
- **Timestamps**: `REAL NOT NULL` columns store Unix epoch floats. Booleans are `INTEGER NOT NULL DEFAULT 0`.
- **No FOREIGN KEY constraints** anywhere in the project schema.

```python
# services/veil_models.py
"""Veil cog data models."""
from __future__ import annotations

from dataclasses import dataclass, field
from typing import Optional


@dataclass
class BoundingBox:
    x1: float
    y1: float
    x2: float
    y2: float

    @property
    def width(self) -> float:
        return self.x2 - self.x1

    @property
    def height(self) -> float:
        return self.y2 - self.y1

    @property
    def area(self) -> float:
        return max(0.0, self.width) * max(0.0, self.height)


@dataclass
class Detection:
    label: str
    score: float
    box: BoundingBox


@dataclass
class VeilConfig:
    guild_id: int
    veil_role_id: int = 0

    reuse_quiet_hours: int = 24
```

```python
# services/veil_pipeline.py
"""Veil pipeline — pure geometry helpers and pipeline orchestrator.

Import this module freely; NudeNet / mediapipe / PIL are imported lazily
inside run_pipeline() only.
"""
from __future__ import annotations

from pathlib import Path

from services.veil_models import BoundingBox, Detection, PipelineResult


DIFFICULTY_PADDING: dict[str, float] = {
    "easy": 0.60,
    "medium": 0.25,
    "hard": 0.05,
}


def iou(box_a: BoundingBox, box_b: BoundingBox) -> float:
    """Intersection over Union of two bounding boxes. Returns 0.0 on no overlap."""
    ix1 = max(box_a.x1, box_b.x1)
    iy1 = max(box_a.y1, box_b.y1)
    ix2 = min(box_a.x2, box_b.x2)
    iy2 = min(box_a.y2, box_b.y2)
    intersection = max(0.0, ix2 - ix1) * max(0.0, iy2 - iy1)
    if intersection == 0.0:
        return 0.0
    union = box_a.area + box_b.area - intersection
    return intersection / union if union > 0.0 else 0.0


def compute_padded_crop(
    bbox: BoundingBox,
    difficulty: str,
    img_w: int,
    img_h: int,
) -> BoundingBox:
    """Apply difficulty-tuned padding and clamp to image bounds.

    Padding fraction applies to the larger dimension of the input bbox.

    Raises:
        ValueError: If difficulty is not "easy", "medium", or "hard".
    """
    if difficulty not in DIFFICULTY_PADDING:
        raise ValueError(
            f"Unknown difficulty {difficulty!r}; expected one of {list(DIFFICULTY_PADDING)}"
        )
    pad = DIFFICULTY_PADDING[difficulty] * max(bbox.width, bbox.height)
    return BoundingBox(
        max(0.0, bbox.x1 - pad),
        max(0.0, bbox.y1 - pad),
        min(float(img_w), bbox.x2 + pad),
        min(float(img_h), bbox.y2 + pad),
    )


def filter_candidates(
    detections: list[Detection],
    face_boxes: list[BoundingBox],
    *,
    fallback: bool = True,
    iou_threshold: float = 0.1,
) -> list[Detection]:
    """Remove detections with face IoU above threshold.

    If all candidates are eliminated and fallback=True, returns the single
    highest-score detection so the crop renderer can clip to non-face area.
    """
    def _overlaps_face(det: Detection) -> bool:
        return any(iou(det.box, fb) > iou_threshold for fb in face_boxes)

    filtered = [d for d in detections if not _overlaps_face(d)]
    if not filtered and fallback and detections:
        return [detections[0]]
    return filtered


def enforce_min_size(bbox: BoundingBox, min_px: int = 200) -> BoundingBox:
    """Expand bbox symmetrically so both width and height are >= min_px.

    Does not clamp to image bounds — caller should clamp if needed.
    Center of the original bbox is preserved.
    """
    cx = (bbox.x1 + bbox.x2) / 2.0
    cy = (bbox.y1 + bbox.y2) / 2.0
    half_w = max(bbox.width, float(min_px)) / 2.0
    half_h = max(bbox.height, float(min_px)) / 2.0
    return BoundingBox(cx - half_w, cy - half_h, cx + half_w, cy + half_h)
```

```python
"""NudeNet ONNX wrapper for the Veil cog.

The NudeDetector is loaded lazily on first call to detect().
Importing this module never triggers a model load.
"""
from __future__ import annotations

import logging
from pathlib import Path

from services.veil_models import BoundingBox, Detection

log = logging.getLogger("dungeonkeeper.veil.nudenet")

_MIN_SCORE: float = 0.5
_DETECTOR = None


def get_detector():
    global _DETECTOR
    if _DETECTOR is None:
        from nudenet import NudeDetector  # type: ignore[import-untyped]
        log.info("Loading NudeNet ONNX model (one-time startup cost)...")
        _DETECTOR = NudeDetector()
        log.info("NudeNet model loaded.")
    return _DETECTOR


def detect(image_path: str | Path) -> list[Detection]:
    """Run NudeNet on an image file and return detections with score >= 0.5, sorted descending."""
    raw: list[dict] = get_detector().detect(str(image_path))
    results: list[Detection] = []
    for item in raw:
        score = float(item.get("score", 0.0))
        if score < _MIN_SCORE:
            continue
        box = item["box"]  # [x1, y1, x2, y2]
        results.append(Detection(
            label=str(item.get("class", "")),
            score=score,
            box=BoundingBox(float(box[0]), float(box[1]), float(box[2]), float(box[3])),
        ))
    results.sort(key=lambda d: d.score, reverse=True)
    return results
```

```python
"""Mediapipe face detection wrapper for the Veil cog.

Importing this module is always safe; mediapipe is imported lazily inside detect_faces().
"""
from __future__ import annotations

import logging
from pathlib import Path

from services.veil_models import BoundingBox

log = logging.getLogger("dungeonkeeper.veil.face_detector")

_DETECTOR = None


def _get_detector():
    global _DETECTOR
    if _DETECTOR is None:
        import mediapipe as mp  # type: ignore[import-untyped]
        _DETECTOR = mp.solutions.face_detection.FaceDetection(
            model_selection=1,
            min_detection_confidence=0.5,
        )
        log.info("Mediapipe face detector loaded.")
    return _DETECTOR


def detect_faces(image_path: str | Path) -> list[BoundingBox]:
    """Return face bounding boxes in pixel coords. Returns [] on any failure."""
    import numpy as np  # type: ignore[import-untyped]
    from PIL import Image  # type: ignore[import-untyped]

    try:
        img = Image.open(str(image_path)).convert("RGB")
    except Exception as exc:
        log.warning("Face detector: failed to open %s: %s", image_path, exc)
        return []

    img_array = np.array(img)
    h, w = img_array.shape[:2]

    try:
        result = _get_detector().process(img_array)
    except Exception as exc:
        log.warning("Face detector: mediapipe failed: %s", exc)
        return []

    if not result.detections:
        return []

    boxes: list[BoundingBox] = []
    for det in result.detections:
        bb = det.location_data.relative_bounding_box
        boxes.append(BoundingBox(
            x1=bb.xmin * w,
            y1=bb.ymin * h,
            x2=(bb.xmin + bb.width) * w,
            y2=(bb.ymin + bb.height) * h,
        ))
    return boxes
```

```python
        JPEG bytes.

    Raises:
        ValueError: If bbox degenerates to zero area after clamping.
    """
    import io
    from PIL import Image  # type: ignore[import-untyped]

    img = Image.open(str(image_path)).convert("RGB")
    img_w, img_h = img.size

    x1 = max(0, int(bbox.x1))
    y1 = max(0, int(bbox.y1))
    x2 = min(img_w, int(bbox.x2))
    y2 = min(img_h, int(bbox.y2))

    if x2 <= x1 or y2 <= y1:
        raise ValueError(f"Degenerate crop bbox after clamp: ({x1},{y1})-({x2},{y2})")

    cropped = img.crop((x1, y1, x2, y2))
    cw, ch = cropped.size

    if cw < MIN_OUTPUT_PX or ch < MIN_OUTPUT_PX:
        scale = MIN_OUTPUT_PX / min(cw, ch)
        cropped = cropped.resize((max(MIN_OUTPUT_PX, int(cw * scale)), max(MIN_OUTPUT_PX, int(ch * scale))), Image.BICUBIC)

    buf = io.BytesIO()
    cropped.save(buf, format="JPEG", quality=JPEG_QUALITY)
    jpeg_bytes = buf.getvalue()

    cache_dir.mkdir(parents=True, exist_ok=True)
    (cache_dir / f"{round_id}.jpg").write_bytes(jpeg_bytes)
    log.debug("Saved crop to %s/%s.jpg (%d bytes)", cache_dir, round_id, len(jpeg_bytes))

    return jpeg_bytes
```

```python
# ---------------------------------------------------------------------------
# Pipeline orchestrator — NudeNet / mediapipe / PIL imported lazily
# ---------------------------------------------------------------------------

def run_pipeline(
    image_path: str | Path,
    *,
    difficulty: str = "medium",
    cache_dir: Path,
    round_id: int,
) -> PipelineResult:
    """Run the 4-stage veil crop pipeline on a saved image file.

    Imports nudenet, mediapipe, and PIL lazily; safe to import this module
    without those packages installed.

    The original image at image_path is read-only; it is not deleted here.
    The crop is saved to cache_dir/<round_id>.jpg.

    Returns:
        PipelineResult with all surviving candidates and crops[0] as the first
        JPEG crop bytes. Additional candidates are accessible for re-roll via
        run_reroll().

    Raises:
        ValueError: If no candidates survive detection + face exclusion.
    """
    from PIL import Image as _PIL  # type: ignore[import-untyped]
    from services.veil_nudenet import detect as _nudenet_detect
    from services.veil_face_detector import detect_faces as _detect_faces
    from services.veil_crop_renderer import render_crop as _render_crop

    image_path = Path(image_path)

    candidates = _nudenet_detect(image_path)
    if not candidates:
        raise ValueError("No NudeNet detections above threshold.")

    face_boxes = _detect_faces(image_path)
    candidates = filter_candidates(candidates, face_boxes, fallback=True)
    if not candidates:
        raise ValueError("All candidates eliminated and fallback failed.")

    with _PIL.open(str(image_path)) as _img:
        img_w, img_h = _img.size

    top = candidates[0]
    padded = compute_padded_crop(top.box, difficulty, img_w, img_h)
    padded = enforce_min_size(padded)

    jpeg = _render_crop(image_path, padded, cache_dir=cache_dir, round_id=round_id)
    return PipelineResult(candidates=candidates, crops=[jpeg])


def run_reroll(
    image_path: str | Path,
    candidates: list[Detection],
    candidate_index: int,
    *,
    difficulty: str = "medium",
    cache_dir: Path,
    round_id: int,
) -> bytes:
    """Render the crop for candidates[candidate_index] and return JPEG bytes.

    Used for the "Re-roll crop" button in Phase 2. candidate_index is the
    1-based next candidate (0 is already rendered by run_pipeline).

    Raises:
        IndexError: If candidate_index >= len(candidates).
    """
    from PIL import Image as _PIL  # type: ignore[import-untyped]
    from services.veil_crop_renderer import render_crop as _render_crop

    image_path = Path(image_path)
    det = candidates[candidate_index]

    with _PIL.open(str(image_path)) as _img:
        img_w, img_h = _img.size

    padded = compute_padded_crop(det.box, difficulty, img_w, img_h)
    padded = enforce_min_size(padded)
    return _render_crop(image_path, padded, cache_dir=cache_dir, round_id=round_id)
```
